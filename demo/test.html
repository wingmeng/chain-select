  el: '#select1',
  alias: {
    value: 'id',
    label: 'info.name',
    name: 'info.type',
    children: 'items'
  },

    建议使用异步加载数据，加载完成后再赋值给组件

var select2;

$.getJSON('./test-data.json', function(res) {
  select2 = new ChainSelect({
    el: '#type',
    alias: {
      value: 'info.areaCode',
      label: 'info.areaName',
      children: 'subs'
    },
    data: res
  });

  // JS 赋值
  $('.btn-primary').prop('disabled', false).click(function() {
    // 中国 广东 东莞 市辖区
    select2.setValues(['CN', 'CN-440000', 'CN-441900', 'CN-441901']);
  });
});
